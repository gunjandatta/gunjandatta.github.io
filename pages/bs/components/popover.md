### TypeScript

```ts
import { Components } from "gd-sprest-bs";

// Create the popover
let el = document.querySelector("#popover");
let popover = Components.Popover({
    el: el,
    isDismissible: true,
    btnProps: {
        text: "Popover Demo"
    },
    options: {
        content: "This is the popover content.",
        title: "My Popover",
        trigger: "hover"
    }
});
```
